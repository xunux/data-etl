CREATE TABLE `report` (
	`health_report_id` BIGINT(20) NOT NULL AUTO_INCREMENT,
	`user_id` VARCHAR(50) NOT NULL,
	`id_card_no_md5` VARCHAR(100) NULL DEFAULT NULL,
	`birthday` DATETIME NULL DEFAULT NULL,
	`sex` VARCHAR(3) NULL DEFAULT NULL,
	`check_unit_code` VARCHAR(100) NULL DEFAULT NULL,
	`check_unit_name` VARCHAR(100) NULL DEFAULT NULL,
	`report_content` MEDIUMTEXT NOT NULL,
	`check_date` DATETIME NULL DEFAULT NULL,
	`last_update_time` DATETIME NULL DEFAULT NULL
	PRIMARY KEY (`health_report_id`)
)
COLLATE='utf8_general_ci'
ENGINE=InnoDB;



CREATE TABLE `user` (
	`user_id` VARCHAR(50) NOT NULL,
	`mobile` VARCHAR(18) NULL DEFAULT NULL,
	`sex` VARCHAR(3) NULL DEFAULT NULL,
	`is_married` TINYINT(4) NULL DEFAULT NULL,
	`has_born` TINYINT(4) NULL DEFAULT NULL,
	`device_model` VARCHAR(40) NULL DEFAULT NULL,
	`city` VARCHAR(40) NULL DEFAULT NULL,
	`last_update_time` DATETIME NULL DEFAULT NULL,
	`birthday` DATETIME NULL DEFAULT NULL,
    `id_card_no_md5` VARCHAR(100) NULL DEFAULT NULL,
	PRIMARY KEY (`user_id`)
)
COLLATE='utf8_general_ci'
ENGINE=InnoDB;
